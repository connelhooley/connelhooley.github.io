using System;

namespace Modality.Node.Shared.Models
{
    public class DialPlanInstruction
    {
        public DialPlanStep Step { get; }
        public string Destination { get; }
        public TimeSpan Duration { get; }
        public DialPlanInstructionType DialPlanInstructionType { get; }

        public DialPlanInstruction(DialPlanStep step, string destination, TimeSpan duration, DialPlanInstructionType dialPlanInstructionType)
        {
            Step = step;
            Destination = destination;
            Duration = duration;
            DialPlanInstructionType = dialPlanInstructionType;
        }
    }
}
